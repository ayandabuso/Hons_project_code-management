We are interested in knowing the total abundance of each insect group recorded in thetaxa colomn which is iportant in addressing the question, the code below answeres that.

taxa_abundance <- insect_data %>%
  group_by(insect_taxa) %>%
  summarise(total_abundance = sum(count, na.rm = TRUE)) %>%
  arrange(desc(total_abundance))

# View summarized table

lets visualise these abundance data!

#first we need to load ggplot2 for this visualization.
library(ggplot2)
# Create a bar plot of insect abundance
ggplot(taxa_abundance, aes(x = reorder(insect_taxa, -total_abundance), y = total_abundance, fill = insect_taxa)) +
  geom_bar(stat = "identity") +
  labs(title = "Total Abundance of Each Insect Taxa",
       x = "Insect Taxa",
       y = "Total Count") +
  theme_minimal() +
  theme(legend.position = "none") +
  coord_flip()

We can also visualize insect abundance between sites

# Summarize abundance per insect taxa at each site
taxa_site_abundance <- insect_data %>%
  group_by(site_name, insect_taxa) %>%
  summarise(total_abundance = sum(count, na.rm = TRUE), .groups = "drop")

# Generate grouped bar plot
ggplot(taxa_site_abundance, aes(x = site_name, y = total_abundance, fill = insect_taxa)) +

We can compare species richness between sites:

# Calculate species richness by counting unique taxa per site
site_species_richness <- insect_data %>%
  group_by(site_name) %>%
  summarise(species_richness = n_distinct(insect_taxa)) %>%
  arrange(desc(species_richness))

# View results
